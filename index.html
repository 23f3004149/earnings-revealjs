Risk Engine – Simplified Provisioning Logic


def compute_expected_loss(exposure, pd, lgd, discount_rate, t_years):
    """
    exposure: outstanding loan amount
    pd: probability of default
    lgd: loss given default
    discount_rate: annual discount rate
    t_years: time horizon (years)
    """
    raw_el = exposure * pd * lgd
    discount_factor = (1 + discount_rate) ** t_years
    return raw_el / discount_factor

el_pv = compute_expected_loss(10_00_00_000, 0.03, 0.40, 0.08, 1)
print(f"Expected loss (PV): ₹{el_pv:,.2f}")
